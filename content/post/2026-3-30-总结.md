# dp(以牛客1037D)
- 设第i步染色
- 当可以使用O(n²)可以使用dp
- 可以从0-(i-1)开始染色
``` cpp {lineNumbers="true"}
#include<bits/stdc++.h>
using namespace std;
#define int long long
void solve(){
    int n;
    cin>>n;
    vector<int>arr(n+1,0),s(n+1,0);
    for(int i=1;i<=n;i++){
        cin>>arr[i];
        s[i]=s[i-1]^arr[i];
    }
    vector<int>dp(n+1);
    dp[0]=0;
    for(int i=1;i<=n;i++){
        dp[i]=dp[i-1]+arr[i];
        int m=1e18;
        for(int j=i;j>=1;j--){
            m=min(m,dp[j-1]);
            dp[i]=min(dp[i],m+(s[i]^s[j-1]));
        }
    }
    cout<<dp[n]<<endl;
}
signed main(){
    int t;
    cin>>t;
    while(t--){
        solve();
    }
}
```
